### Q1. 신규/정기 결제 건수 Monthly

```sql
SELECT
    DATE_FORMAT(GPI.UPDATE_DATE, "%Y-%m") AS 결제월,
    SUM(CASE WHEN GPI.PAYMENT_DIV = 'F' AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) AS 신규결제,
    SUM(CASE WHEN GPI.PAYMENT_DIV = 'S' AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) AS 정기결제,
    SUM(CASE WHEN GPI.PAYMENT_DIV IN ('F', 'S') AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) AS total,
    ROUND(SUM(CASE WHEN GPI.PAYMENT_DIV = 'F' AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) / SUM(CASE WHEN GPI.PAYMENT_DIV IN ('F', 'S') AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) * 100, 1) AS 신규비율,
    ROUND(SUM(CASE WHEN GPI.PAYMENT_DIV = 'S' AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) / SUM(CASE WHEN GPI.PAYMENT_DIV IN ('F', 'S') AND GPI.PAID_AMOUNT > 0 THEN 1 ELSE 0 END) * 100, 1) AS 정기비율
FROM GT_PAYMENT_INFO GPI
WHERE GPI.CLASS_TYPE = 'PODO'
  AND DATE(GPI.UPDATE_DATE) >= DATE_SUB(CURDATE(), INTERVAL 12 MONTH)
GROUP BY DATE_FORMAT(GPI.UPDATE_DATE, "%Y-%m")
ORDER BY 결제월
```

### Q3. 정기결제 회차별 이탈 분포

```sql
-- 이탈 판정: 마지막 정기결제로부터 45일 이상 경과 = churned
WITH user_sub AS (
    SELECT
        USER_UID,
        COUNT(*) AS sub_count,
        MIN(UPDATE_DATE) AS first_sub_date,
        MAX(UPDATE_DATE) AS last_sub_date,
        DATEDIFF(CURDATE(), MAX(UPDATE_DATE)) AS days_since_last
    FROM GT_PAYMENT_INFO
    WHERE CLASS_TYPE = 'PODO'
      AND PAYMENT_DIV = 'S'
      AND PAID_AMOUNT > 0
    GROUP BY USER_UID
),
user_status AS (
    SELECT *,
        CASE WHEN days_since_last > 45 THEN 'churned' ELSE 'active' END AS status
    FROM user_sub
)
SELECT
    sub_count AS 정기결제_회차,
    COUNT(*) AS 유저수,
    SUM(CASE WHEN status = 'churned' THEN 1 ELSE 0 END) AS 이탈수,
    SUM(CASE WHEN status = 'active' THEN 1 ELSE 0 END) AS 유지수,
    ROUND(SUM(CASE WHEN status = 'churned' THEN 1 ELSE 0 END) / COUNT(*) * 100, 1) AS 이탈률
FROM user_status
WHERE sub_count <= 15
GROUP BY sub_count
ORDER BY sub_count
```

### Q4. 코호트별 정기전환 잔존율

```sql
-- 첫 신규결제 월 기준, N개월 후 정기결제 여부
WITH first_pay AS (
    SELECT
        USER_UID,
        MIN(UPDATE_DATE) AS first_pay_date,
        DATE_FORMAT(MIN(UPDATE_DATE), '%Y-%m') AS cohort_month
    FROM GT_PAYMENT_INFO
    WHERE CLASS_TYPE = 'PODO'
      AND PAYMENT_DIV = 'F'
      AND PAID_AMOUNT > 0
    GROUP BY USER_UID
),
sub_months AS (
    SELECT
        USER_UID,
        DATE_FORMAT(UPDATE_DATE, '%Y-%m') AS sub_month
    FROM GT_PAYMENT_INFO
    WHERE CLASS_TYPE = 'PODO'
      AND PAYMENT_DIV = 'S'
      AND PAID_AMOUNT > 0
    GROUP BY USER_UID, DATE_FORMAT(UPDATE_DATE, '%Y-%m')
),
cohort_data AS (
    SELECT
        fp.cohort_month,
        fp.USER_UID,
        PERIOD_DIFF(EXTRACT(YEAR_MONTH FROM STR_TO_DATE(sm.sub_month, '%Y-%m')),
                     EXTRACT(YEAR_MONTH FROM fp.first_pay_date)) AS months_after
    FROM first_pay fp
    JOIN sub_months sm ON fp.USER_UID = sm.USER_UID
    WHERE fp.cohort_month >= '2025-01'
)
SELECT
    cohort_month AS 코호트,
    COUNT(DISTINCT CASE WHEN months_after = 1 THEN USER_UID END) AS M1,
    COUNT(DISTINCT CASE WHEN months_after = 2 THEN USER_UID END) AS M2,
    COUNT(DISTINCT CASE WHEN months_after = 3 THEN USER_UID END) AS M3,
    COUNT(DISTINCT CASE WHEN months_after = 4 THEN USER_UID END) AS M4,
    COUNT(DISTINCT CASE WHEN months_after = 5 THEN USER_UID END) AS M5,
    COUNT(DISTINCT CASE WHEN months_after = 6 THEN USER_UID END) AS M6,
    COUNT(DISTINCT CASE WHEN months_after >= 1 THEN USER_UID END) AS 정기전환_총유저
FROM cohort_data
GROUP BY cohort_month
ORDER BY cohort_month
```
